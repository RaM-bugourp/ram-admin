## 开发步骤

### 步骤 1：API 客户端（Axios 封装 + CSRF）

文件：`front-end/src/api/client.ts`

```typescript
import axios, { AxiosError, InternalAxiosRequestConfig } from 'axios'
import router from '@/router'
import store from '@/stores'

const client = axios.create({
    baseURL: import.meta.env.VITE_API_BASE_URL || '/api',
    timeout: 15000,
    headers: { 'Content-Type': 'application/json' },
    withCredentials: true,  // Session认证：自动携带Cookie
    xsrfCookieName: 'csrftoken',
    xsrfHeaderName: 'X-CSRFToken',
})

// 请求拦截器：注入 CSRF Token
client.interceptors.request.use((config: InternalAxiosRequestConfig) => {
    // Django CSRF token由浏览器自动从cookie读取并通过header发送
    // withCredentials: true + xsrfCookieName/xsrfHeaderName 配置已处理
    return config
})

// 响应拦截器：统一错误处理
client.interceptors.response.use(
    (response) => response.data,  // 解包 { data: ... } 包装
    (error: AxiosError<ApiError>) => {
        const status = error.response?.status
        if (status === 401) {
            store.dispatch('user/logout')
            router.push('/login')
        }
        if (status === 403) {
            router.push('/403')
        }
        return Promise.reject({
            code: error.response?.data?.error?.code || 'UNKNOWN',
            message: error.response?.data?.error?.message || '请求失败',
            status,
        })
    }
)

export default client

interface ApiError {
    error: { code: string; message: string }
}
```

import type { UserCreateInput, UserOutput } from '../types/user'

export const userApi = {
  list: (params?: { page?: number; page_size?: number }) =>
    client.get<PaginatedResponse<UserOutput>>('/users/', { params }),

  getById: (id: number) =>
    client.get<UserOutput>(`/users/${id}/`),

  create: (data: UserCreateInput) =>
    client.post<UserOutput>('/users/', data),

  update: (id: number, data: Partial<UserCreateInput>) =>
    client.put(`/users/${id}/`, data),

  delete: (id: number) =>
    client.delete(`/users/${id}/`),

  getMe: () =>
    client.get<UserOutput>('/users/me/'),
}

interface PaginatedResponse<T> {
  data: T[]
  pagination: { page: number; page_size: number; total: number }
}

### 步骤 3：路由配置（含权限守卫）

文件：`front-end/src/router/index.ts`

```typescript
import { createRouter, createWebHistory } from 'vue-router'
import store from '@/stores'

const routes = [
  {
    path: '/login',
    component: () => import('@/views/login/LoginView.vue'),
    meta: { requiresAuth: false },
  },
  {
    path: '/',
    component: () => import('@/layouts/MainLayout.vue'),
    meta: { requiresAuth: true },
    children: [
      { path: '', redirect: '/dashboard' },
      {
        path: 'dashboard',
        component: () => import('@/views/dashboard/DashboardView.vue'),
      },
      {
        path: 'users',
        component: () => import('@/views/user/UserListView.vue'),
        meta: { permission: 'user:read' },
      },
      {
        path: 'roles',
        component: () => import('@/views/role/RoleListView.vue'),
        meta: { permission: 'role:read' },
      },
    ],
  },
]

const router = createRouter({
  history: createWebHistory(),
  routes,
})

router.beforeEach((to, _from, next) => {
  if (to.meta.requiresAuth !== false && !store.getters['user/isAuthenticated']) {
    return next('/login')
  }
  if (to.meta.permission && !store.getters['user/hasPermission'](to.meta.permission as string)) {
    return next('/403')
  }
  next()
})

export default router
```

文件：`front-end/src/stores/user.ts`

```typescript
import { Module } from 'vuex'
import { userApi } from '@/api/modules/user'
import type { UserOutput } from '@/api/types/user'

interface UserState {
    currentUser: UserOutput | null
    permissions: string[]
}

const user: Module<UserState, any> = {
    namespaced: true,
    state: (): UserState => ({
        currentUser: null,
        permissions: [],
    }),
    getters: {
        isAuthenticated(state): boolean {
            return !!state.currentUser
        },
        isAdmin(state): boolean {
            return state.currentUser?.roles?.some(
                (r: any) => r.name === 'admin'
            ) ?? false
        },
    },
    mutations: {
        SET_USER(state, user: UserOutput) {
            state.currentUser = user
        },
        SET_PERMISSIONS(state, permissions: string[]) {
            state.permissions = permissions
        },
        CLEAR(state) {
            state.currentUser = null
            state.permissions = []
        },
    },
    actions: {
        hasPermission({ state, getters }, code: string): boolean {
            return getters.isAdmin || state.permissions.includes(code)
        },
        async login({ dispatch }, { username, password }) {
            await userApi.login({ username, password })
            await dispatch('fetchCurrentUser')
        },
        async fetchCurrentUser({ commit }) {
            const user = await userApi.getMe()
            commit('SET_USER', user)
        },
        async logout({ commit }) {
            await userApi.logout()
            commit('CLEAR')
        },
    },
}

export default user
```

### 步骤 5：通用组合函数

文件：`front-end/src/composables/useRequest.ts`

```typescript
import { ref } from 'vue'

export function useRequest<T>(fetcher: () => Promise<T>) {
  const data = ref<T | null>(null)
  const loading = ref(false)
  const error = ref<string | null>(null)

  async function execute() {
    loading.value = true
    error.value = null
    try {
      data.value = await fetcher()
    } catch (e: any) {
      error.value = e.message || '请求失败'
    } finally {
      loading.value = false
    }
  }

  return { data, loading, error, execute }
}
```
