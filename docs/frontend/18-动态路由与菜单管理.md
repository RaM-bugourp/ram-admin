# 18 - 动态路由与菜单管理

## 适用场景
管理后台的菜单通过数据库配置，前端根据后端返回的菜单树动态注册路由。新增页面无需修改前端代码重新构建。

## 关联原则
- 原则6：最小权限 — 菜单可见性基于角色权限
- 原则2：契约先行 — 菜单树 API 格式是前后端契约的典型案例

## 核心规则
1. **菜单数据存在数据库**，后端返回用户可见的菜单树
2. **前端根据菜单树动态注册路由**（Vite `import.meta.glob` + `router.addRoute`）
3. **菜单与权限绑定**：菜单项关联权限码，无权限不展示
4. **路由懒加载**：动态路由也必须支持 code-splitting

## 开发步骤

### 步骤 1：菜单 Model

文件：`apps/rbac/models.py` — Menu 模型

```python
from django.db import models
from apps.common.base_model import BaseModel
from apps.common.managers import BaseManager


class Menu(BaseModel):
    """菜单 — 支持多级树结构"""
    MENU_TYPE_CHOICES = [
        ('directory', '目录'),    # 可展开的父级
        ('menu', '菜单'),         # 可点击的页面
        ('button', '按钮'),       # 页面内的操作按钮
    ]

    parent = models.ForeignKey(
        'self', on_delete=models.CASCADE, null=True, blank=True,
        related_name='children', verbose_name="父菜单"
    )
    name = models.CharField(max_length=50, verbose_name="菜单名称")
    path = models.CharField(max_length=200, default="", verbose_name="路由路径")
    component = models.CharField(max_length=200, default="", verbose_name="组件路径")
    icon = models.CharField(max_length=50, default="", verbose_name="图标")
    menu_type = models.CharField(
        max_length=20, choices=MENU_TYPE_CHOICES, default='menu',
        verbose_name="菜单类型"
    )
    permission_code = models.CharField(
        max_length=100, default="", verbose_name="权限编码（为空则公开）"
    )
    sort = models.IntegerField(default=0, verbose_name="排序")
    visible = models.BooleanField(default=True, verbose_name="是否可见")

    objects = BaseManager()

    class Meta:
        db_table = 'rbac_menus'
        verbose_name = '菜单'
        ordering = ['sort']

    @classmethod
    def get_user_menu_tree(cls, user) -> list:
        """获取用户可见的菜单树"""
        menus = cls.objects.filter(visible=True).select_related('parent')

        if not user.is_superuser:
            # 获取用户所有权限码
            user_codes = set(
                Permission.objects.filter(
                    rolepermission__role__userrole__user=user
                ).values_list('code', flat=True)
            )
            # 过滤：权限码为空（公开菜单）或用户拥有该权限码
            menus = [
                m for m in menus
                if not m.permission_code or m.permission_code in user_codes
            ]

        return cls._build_tree(menus)

    @classmethod
    def _build_tree(cls, menus, parent_id=None):
        """递归构建菜单树"""
        tree = []
        for menu in menus:
            if menu.parent_id == parent_id:
                node = {
                    'id': menu.id,
                    'name': menu.name,
                    'path': menu.path,
                    'component': menu.component,
                    'icon': menu.icon,
                    'menu_type': menu.menu_type,
                    'permission_code': menu.permission_code,
                    'children': cls._build_tree(menus, menu.id),
                }
                tree.append(node)
        return tree
```

### 步骤 2：菜单树 API

文件：`apps/rbac/views/menu_views.py`

```python
from rest_framework import viewsets
from rest_framework.response import Response
from rest_framework.permissions import IsAuthenticated
from apps.rbac.models import Menu


class MenuViewSet(viewsets.GenericViewSet):
    """菜单接口"""
    permission_classes = [IsAuthenticated]

    def list(self, request):
        """返回当前用户的菜单树"""
        tree = Menu.get_user_menu_tree(request.user)
        return Response({"data": tree})
```

### 步骤 3：前端动态路由注册

文件：`front-end/src/router/dynamic-routes.ts`

```typescript
import type { RouteRecordRaw } from 'vue-router'
import MainLayout from '@/layouts/MainLayout.vue'

// Vite 静态分析：用 import.meta.glob 匹配所有页面组件
const pageModules = import.meta.glob('@/views/**/*.vue')

interface MenuItem {
    id: number
    name: string
    path: string
    component: string
    icon: string
    menu_type: 'directory' | 'menu' | 'button'
    permission_code: string
    children: MenuItem[]
}

/**
 * 将后端菜单树转换为 Vue Router 路由配置
 */
export function buildRoutes(menuTree: MenuItem[]): RouteRecordRaw[] {
    return menuTree
        .filter(m => m.menu_type !== 'button')  // 按钮不生成路由
        .map(menu => {
            const route: RouteRecordRaw = {
                path: menu.path,
                name: menu.name,
                meta: {
                    title: menu.name,
                    icon: menu.icon,
                    permission: menu.permission_code || undefined,
                },
                redirect: menu.menu_type === 'directory'
                    ? menu.children?.[0]?.path
                    : undefined,
                component: menu.menu_type === 'directory'
                    ? MainLayout
                    : resolveComponent(menu.component),
            }

            if (menu.children?.length) {
                route.children = buildRoutes(menu.children)
            }

            return route
        })
}

/**
 * 根据 component 路径匹配 Vite 动态导入
 * 例如 component="system/UserListView" → glob key "@/views/system/UserListView.vue"
 */
function resolveComponent(componentPath: string) {
    const key = `/src/views/${componentPath}.vue`
    const loader = pageModules[key]

    if (!loader) {
        console.warn(`[Router] 找不到组件: ${key}，使用 404 降级`)
        return () => import('@/views/error/NotFoundView.vue')
    }

    return loader  // 返回 () => import(...) 函数，支持懒加载
}
```

### 步骤 4：路由初始化流程

```typescript
// front-end/src/router/index.ts
import { createRouter, createWebHistory } from 'vue-router'
import store from '@/stores'
import { buildRoutes } from './dynamic-routes'
import { menuApi } from '@/api/modules/menu'

// 静态路由（不受权限控制）
const staticRoutes: RouteRecordRaw[] = [
    {
        path: '/login',
        component: () => import('@/views/login/LoginView.vue'),
        meta: { requiresAuth: false },
    },
    {
        path: '/403',
        component: () => import('@/views/error/ForbiddenView.vue'),
        meta: { requiresAuth: false },
    },
    {
        path: '/:pathMatch(.*)*',
        component: () => import('@/views/error/NotFoundView.vue'),
    },
]

const router = createRouter({
    history: createWebHistory(),
    routes: staticRoutes,
})

// 标记是否已加载动态路由（防止重复加载）
let dynamicRoutesLoaded = false

router.beforeEach(async (to, _from, next) => {
    // 跳过登录页
    if (to.path === '/login') return next()

    // 检查认证状态
    if (!store.getters['user/isAuthenticated']) {
        return next('/login')
    }

    // 首次加载：获取菜单树并注册动态路由
    if (!dynamicRoutesLoaded) {
        try {
            const menuTree = await menuApi.getUserMenu()
            const routes = buildRoutes(menuTree)
            routes.forEach(route => router.addRoute(route))
            dynamicRoutesLoaded = true
            // 重新导航到目标路由（此时动态路由已注册）
            return next({ ...to, replace: true })
        } catch {
            return next('/login')
        }
    }

    next()
})

export default router
```

```vue
<template>
    <a-menu
        :selected-keys="[currentPath]"
        @menu-item-click="navigate"
        :collapsed="collapsed"
    >
        <template v-for="menu in menuTree" :key="menu.id">
            <a-sub-menu v-if="menu.children?.length" :key="menu.id">
                <template #icon><component :is="menu.icon" /></template>
                <template #title>{{ menu.name }}</template>
                <a-menu-item
                    v-for="child in menu.children"
                    :key="child.id"
                >
                    {{ child.name }}
                </a-menu-item>
            </a-sub-menu>
            <a-menu-item v-else :key="menu.id">
                <template #icon><component :is="menu.icon" /></template>
                {{ menu.name }}
            </a-menu-item>
        </template>
    </a-menu>
</template>
```

## 反模式（禁止）
- ❌ 前端硬编码路由表 → 从后端菜单树动态生成
- ❌ 菜单和权限分离 → 菜单项关联 permission_code
- ❌ 动态路由不使用懒加载 → `import.meta.glob` 支持 code-splitting
- ❌ 忘记降级处理 → 找不到组件时跳404
