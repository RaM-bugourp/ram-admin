# 14 - 前端页面与组件开发

## 适用场景
API 层和类型就绪后，开发具体的页面和组件。

## 关联原则
- 原则1：SRP — 一个组件只做一件事
- 原则3：关注点分离 — 页面（路由级）/ 业务组件 / UI 组件严格分层
- 原则5：输出转义 — 防 XSS

## 组件分层

```
views/          ← 页面级（路由挂载），组合 components + composables
components/
  └── business/ ← 业务组件（有领域语义：UserForm, RoleTree, PermissionPanel）
  └── common/   ← UI 组件（无领域语义：Table, Modal, SearchBar）
composables/    ← 逻辑组合（无 UI：useTable, useForm, usePermission）
```

## 开发步骤

### 步骤 1：页面组件模板（列表页）

文件：`front-end/src/views/user/UserListView.vue`

```vue
<template>
  <div class="user-list-page">
    <!-- 搜索栏 -->
    <SearchBar v-model="searchQuery" @search="handleSearch" />

    <!-- 操作栏 -->
    <div class="toolbar">
      <el-button v-if="hasPermission('user:create')" type="primary" @click="showCreateDialog = true">
        新建用户
      </el-button>
    </div>

    <!-- 表格 -->
    <el-table :data="tableData" v-loading="loading" border stripe>
      <el-table-column prop="id" label="ID" width="80" />
      <el-table-column prop="username" label="用户名" />
      <el-table-column prop="email" label="邮箱" />
      <el-table-column prop="is_active" label="状态" width="100">
        <template #default="{ row }">
          <el-tag :type="row.is_active ? 'success' : 'danger'">
            {{ row.is_active ? '激活' : '禁用' }}
          </el-tag>
        </template>
      </el-table-column>
      <el-table-column label="操作" width="200">
        <template #default="{ row }">
          <el-button size="small" @click="handleEdit(row)">编辑</el-button>
          <el-button size="small" type="danger" @click="handleDelete(row.id)">删除</el-button>
        </template>
      </el-table-column>
    </el-table>

    <!-- 分页 -->
    <Pagination v-model:page="page" v-model:pageSize="pageSize"
                :total="total" @change="fetchData" />

    <!-- 新建/编辑弹窗 -->
    <UserFormDialog v-model:visible="showCreateDialog"
                    :user="editingUser"
                    @success="fetchData" />
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { userApi } from '@/api/modules/user'
import { usePermission } from '@/composables/usePermission'
import type { UserOutput } from '@/api/types/user'

const { hasPermission } = usePermission()

const loading = ref(false)
const tableData = ref<UserOutput[]>([])
const searchQuery = ref('')
const page = ref(1)
const pageSize = ref(20)
const total = ref(0)

const showCreateDialog = ref(false)
const editingUser = ref<UserOutput | null>(null)

async function fetchData() {
  loading.value = true
  try {
    const res = await userApi.list({ page: page.value, page_size: pageSize.value })
    tableData.value = res.data
    total.value = res.pagination.total
  } finally {
    loading.value = false
  }
}

function handleSearch() { page.value = 1; fetchData() }
function handleEdit(user: UserOutput) { editingUser.value = user; showCreateDialog.value = true }
async function handleDelete(id: number) {
  await userApi.delete(id)
  fetchData()
}

onMounted(fetchData)
</script>
```

### 步骤 2：业务组件模板（表单弹窗）

文件：`front-end/src/components/business/UserFormDialog.vue`

```vue
<template>
  <el-dialog :model-value="visible" :title="isEdit ? '编辑用户' : '新建用户'"
             @update:model-value="$emit('update:visible', $event)"
             @close="resetForm">
    <el-form ref="formRef" :model="form" :rules="rules" label-width="80px">
      <el-form-item label="用户名" prop="username">
        <el-input v-model="form.username" maxlength="20" />
      </el-form-item>
      <el-form-item label="邮箱" prop="email">
        <el-input v-model="form.email" />
      </el-form-item>
      <el-form-item v-if="!isEdit" label="密码" prop="password">
        <el-input v-model="form.password" type="password" show-password />
      </el-form-item>
    </el-form>
    <template #footer>
      <el-button @click="$emit('update:visible', false)">取消</el-button>
      <el-button type="primary" :loading="submitting" @click="handleSubmit">确定</el-button>
    </template>
  </el-dialog>
</template>

<script setup lang="ts">
import { ref, reactive, computed, watch } from 'vue'
import { userApi } from '@/api/modules/user'
import { ElMessage } from 'element-plus'
import type { UserOutput, UserCreateInput } from '@/api/types/user'

const props = defineProps<{
  visible: boolean
  user: UserOutput | null
}>()

const emit = defineEmits<{
  'update:visible': [value: boolean]
  success: []
}>()

const isEdit = computed(() => !!props.user)
const submitting = ref(false)

const form = reactive<UserCreateInput>({ username: '', email: '', password: '', role_ids: [] })
const rules = {
  username: [{ required: true, message: '必填' }, { min: 3, max: 20, message: '3-20 位' }],
  email: [{ required: true, type: 'email', message: '请输入有效邮箱' }],
  password: [{ required: true, min: 8, message: '至少 8 位' }],
}

watch(() => props.user, (u) => {
  if (u) Object.assign(form, { username: u.username, email: u.email, password: '' })
}, { immediate: true })

function resetForm() {
  Object.assign(form, { username: '', email: '', password: '', role_ids: [] })
}

async function handleSubmit() {
  submitting.value = true
  try {
    if (isEdit.value) {
      await userApi.update(props.user!.id, form)
    } else {
      await userApi.create(form)
    }
    ElMessage.success(isEdit.value ? '更新成功' : '创建成功')
    emit('success')
    emit('update:visible', false)
  } catch (e: any) {
    ElMessage.error(e.message)
  } finally {
    submitting.value = false
  }
}
</script>
```

### 步骤 3：通用组合函数（useTable）

文件：`front-end/src/composables/useTable.ts`

```typescript
import { ref, reactive } from 'vue'

export function useTable<T>(fetcher: (params: any) => Promise<{ data: T[]; pagination: any }>) {
  const data = ref<T[]>([])
  const loading = ref(false)
  const pagination = reactive({ page: 1, pageSize: 20, total: 0 })

  async function fetch(params = {}) {
    loading.value = true
    try {
      const res = await fetcher({ page: pagination.page, page_size: pagination.pageSize, ...params })
      data.value = res.data
      pagination.total = res.pagination.total
    } finally {
      loading.value = false
    }
  }

  function onPageChange(page: number) { pagination.page = page; fetch() }
  function onPageSizeChange(size: number) { pagination.pageSize = size; pagination.page = 1; fetch() }

  return { data, loading, pagination, fetch, onPageChange, onPageSizeChange }
}
```

## 组件三态处理（强制）

每个数据展示组件必须处理三种状态：

```vue
<template>
  <!-- Loading 态 -->
  <el-skeleton v-if="loading" :rows="5" animated />

  <!-- Empty 态 -->
  <el-empty v-else-if="!data.length" description="暂无数据" />

  <!-- 数据态 -->
  <el-table v-else :data="data">
    ...
  </el-table>
</template>
```

## 反模式（禁止）
- ❌ 页面直接 `axios.get()` → 通过 API 模块
- ❌ 组件承担数据获取 + 展示 + 表单逻辑 → 拆分
- ❌ 不处理 loading/empty/error 三态
- ❌ v-html 未经过 DOMPurify
