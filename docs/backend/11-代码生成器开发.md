# 11 - 代码生成器开发

## 适用场景
根据已有 Model 自动生成标准的 ViewSet / Serializer / Service / Repository 代码，减少重复工作。

## 关联原则
- 原则3：关注点分离 — 生成代码必须遵循分层架构
- 原则2：契约先行 — 自动生成 Serializer 契约

## 核心规则
1. **生成的代码必须符合分层架构**（View → Service → Repository）
2. **生成的是脚手架，不是最终代码**（业务逻辑留 TODO）
3. **覆盖 CRUD 标准场景**

## 开发步骤

### 步骤 1：创建代码生成命令

文件：`apps/codegen/management/commands/generate_module.py`

```python
from django.core.management.base import BaseCommand
from django.apps import apps

CRUD_TEMPLATES = {
    'repository': '''from apps.common.repositories import BaseRepository
from apps.{app_name}.models import {model_name}

class {model_name}Repository(BaseRepository[{model_name}]):
    """{model_name} 数据访问层"""
    model = {model_name}
''',

    'serializer': '''from rest_framework import serializers

class {model_name}CreateSerializer(serializers.Serializer):
    """创建 {model_name} - 输入契约"""
    # TODO: 根据模型字段补充
    pass

    class Meta:
        version = "v1.0.0"

class {model_name}OutputSerializer(serializers.Serializer):
    """{model_name} - 输出契约"""
    id = serializers.IntegerField(read_only=True)
    created_at = serializers.DateTimeField(read_only=True)

    class Meta:
        version = "v1.0.0"
''',

    'service': '''from apps.{app_name}.repositories import {model_name}Repository
from apps.common.exceptions import BusinessError
from django.db import transaction

class {model_name}Service:
    """{model_name} 业务逻辑层"""

    def __init__(self, repo=None):
        self.repo = repo or {model_name}Repository()

    def create(self, data: dict) -> dict:
        """创建 {model_name}"""
        obj = self.repo.create(data)
        return {{'id': obj.id}}

    def get_by_id(self, obj_id):
        obj = self.repo.get_by_id(obj_id)
        if not obj:
            raise BusinessError(
                message="{model_name}不存在", code="NOT_FOUND", status=404
            )
        return obj
''',

    'viewset': '''from rest_framework import viewsets, status
from rest_framework.response import Response
from rest_framework.permissions import IsAuthenticated, IsAdminUser
from apps.{app_name}.services import {model_name}Service
from apps.{app_name}.serializers import (
    {model_name}CreateSerializer,
    {model_name}OutputSerializer,
)

class {model_name}ViewSet(viewsets.GenericViewSet):
    """{model_name} 管理接口"""

    def get_permissions(self):
        return {
            'list': [IsAuthenticated],
            'create': [IsAdminUser],
            'update': [IsAdminUser],
            'destroy': [IsAdminUser],
        }.get(self.action, [IsAuthenticated])

    def list(self, request):
        queryset = {model_name}Service().list_all()
        serializer = {model_name}OutputSerializer(queryset, many=True)
        return Response(serializer.data)

    def create(self, request):
        serializer = {model_name}CreateSerializer(data=request.data)
        serializer.is_valid(raise_exception=True)
        result = {model_name}Service().create(serializer.validated_data)
        return Response(result, status=status.HTTP_201_CREATED)
''',
}

class Command(BaseCommand):
    help = '根据 Model 生成标准 CRUD 代码'

    def add_arguments(self, parser):
        parser.add_argument('app_name', type=str, help='App 名称（如 rbac）')
        parser.add_argument('model_name', type=str, help='Model 名称（如 User）')

    def handle(self, *args, **options):
        app_name = options['app_name']
        model_name = options['model_name']

        context = {'app_name': app_name, 'model_name': model_name}

        # 生成 Repository
        repo_path = f'apps/{app_name}/repositories.py'
        self._append_or_create(repo_path, CRUD_TEMPLATES['repository'].format(**context))
        self.stdout.write(f'✅ Repository → {repo_path}')

        # 生成 Serializer
        ser_path = f'apps/{app_name}/serializers/{model_name.lower()}_serializers.py'
        self._append_or_create(ser_path, CRUD_TEMPLATES['serializer'].format(**context))
        self.stdout.write(f'✅ Serializer → {ser_path}')

        # 生成 Service
        svc_path = f'apps/{app_name}/services/{model_name.lower()}_service.py'
        self._append_or_create(svc_path, CRUD_TEMPLATES['service'].format(**context))
        self.stdout.write(f'✅ Service → {svc_path}')

        # 生成 ViewSet
        view_path = f'apps/{app_name}/views/{model_name.lower()}_views.py'
        self._append_or_create(view_path, CRUD_TEMPLATES['viewset'].format(**context))
        self.stdout.write(f'✅ ViewSet → {view_path}')

    def _append_or_create(self, path, content):
        import os
        os.makedirs(os.path.dirname(path), exist_ok=True)
        with open(path, 'a' if os.path.exists(path) else 'w') as f:
            if os.path.exists(path):
                f.write('\n')
            f.write(content.strip() + '\n')
```

### 步骤 2：使用方式

```bash
# 为一个实体生成全套 CRUD 代码
python manage.py generate_module rbac User
python manage.py generate_module rbac Role
python manage.py generate_module rbac Permission
```

### 步骤 3：生成效果

执行后会生成 4 个文件（如有则追加）：
- `apps/rbac/repositories.py` — Repository 类
- `apps/rbac/serializers/user_serializers.py` — 输入/输出 Serializer
- `apps/rbac/services/user_service.py` — Service 类
- `apps/rbac/views/user_views.py` — ViewSet 类

## 反模式（禁止）
- ❌ 生成代码直接用于生产（必须人工 review）
- ❌ 生成的代码不符合分层架构
- ❌ 生成代码覆盖已有手写代码（只追加）
