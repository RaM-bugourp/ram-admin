# 20 - 错误码体系

## 适用场景
定义项目统一的错误码枚举，供 Service 层抛出、ViewSet 层捕获转换为 HTTP 响应、前端统一处理。

## 关联原则
- 原则5：输入验证 — 所有校验失败都有对应错误码
- 原则3：关注点分离 — 错误码定义集中管理，不在各处散落字符串

## 核心规则
1. **所有业务异常通过 `ErrorCode` 枚举抛出**（不用裸字符串 code）
2. **错误码三要素**：`code`（前端识别）、`message`（默认中文消息）、`http_status`（HTTP 状态码）
3. **按模块分组**：通用 / 用户 / 角色 / 权限 / 审计 / 文件
4. **前端根据 code 做 UI 处理**（如 `USER_NOT_FOUND` → 跳转 404 页）

## 开发步骤

### 步骤 1：定义错误码枚举

文件：`apps/common/exceptions.py`

```python
from enum import Enum
from dataclasses import dataclass


@dataclass(frozen=True)
class ErrorDetail:
    """错误详情"""
    code: str
    message: str
    http_status: int = 400


class ErrorCode(Enum):
    """统一错误码枚举"""

    # ---------- 通用 ----------
    VALIDATION_ERROR = ErrorDetail("VALIDATION_ERROR", "输入数据校验失败", 400)
    NOT_FOUND = ErrorDetail("NOT_FOUND", "资源不存在", 404)
    INTERNAL_ERROR = ErrorDetail("INTERNAL_ERROR", "服务器内部错误", 500)
    PERMISSION_DENIED = ErrorDetail("PERMISSION_DENIED", "权限不足", 403)
    RATE_LIMITED = ErrorDetail("RATE_LIMITED", "请求过于频繁", 429)

    # ---------- 用户 ----------
    USER_NOT_FOUND = ErrorDetail("USER_NOT_FOUND", "用户不存在", 404)
    USERNAME_EXISTS = ErrorDetail("USERNAME_EXISTS", "用户名已存在", 409)
    EMAIL_EXISTS = ErrorDetail("EMAIL_EXISTS", "邮箱已存在", 409)
    INVALID_CREDENTIALS = ErrorDetail("INVALID_CREDENTIALS", "用户名或密码错误", 401)
    USER_INACTIVE = ErrorDetail("USER_INACTIVE", "用户已被禁用", 403)
    ADMIN_REQUIRES_DEPARTMENT = ErrorDetail(
        "ADMIN_REQUIRES_DEPARTMENT", "管理员角色必须指定部门", 400
    )

    # ---------- 角色 ----------
    ROLE_NOT_FOUND = ErrorDetail("ROLE_NOT_FOUND", "角色不存在", 404)
    ROLE_CODE_EXISTS = ErrorDetail("ROLE_CODE_EXISTS", "角色编码已存在", 409)
    ROLE_HAS_USERS = ErrorDetail(
        "ROLE_HAS_USERS", "角色下仍有用户，不可删除", 400
    )

    # ---------- 权限 ----------
    PERMISSION_NOT_FOUND = ErrorDetail("PERMISSION_NOT_FOUND", "权限不存在", 404)
    PERMISSION_CODE_EXISTS = ErrorDetail(
        "PERMISSION_CODE_EXISTS", "权限编码已存在", 409
    )

    # ---------- 审计 ----------
    AUDIT_LOG_FAILED = ErrorDetail("AUDIT_LOG_FAILED", "审计日志写入失败", 500)

    # ---------- 文件 ----------
    FILE_TOO_LARGE = ErrorDetail("FILE_TOO_LARGE", "文件大小超出限制", 413)
    FILE_TYPE_DENIED = ErrorDetail("FILE_TYPE_DENIED", "不支持的文件类型", 415)
    FILE_UPLOAD_FAILED = ErrorDetail("FILE_UPLOAD_FAILED", "文件上传失败", 500)

    # ---------- 菜单 ----------
    MENU_NOT_FOUND = ErrorDetail("MENU_NOT_FOUND", "菜单不存在", 404)
    MENU_PARENT_INVALID = ErrorDetail("MENU_PARENT_INVALID", "无效的父菜单", 400)

    def as_exc(self, detail: str = None):
        """转换为 BusinessError 抛出"""
        err = self.value
        return BusinessError(
            code=err.code,
            message=detail or err.message,
            status=err.http_status,
        )

    @classmethod
    def to_frontend_map(cls) -> dict:
        """生成前端错误处理映射表"""
        return {
            item.value.code: {
                "message": item.value.message,
                "http_status": item.value.http_status,
            }
            for item in cls
        }


class BusinessError(Exception):
    """业务异常"""
    def __init__(self, message: str, code: str, status: int = 400):
        self.message = message
        self.code = code
        self.status = status


def custom_exception_handler(exc, context):
    """DRF 异常处理器 — 统一包装错误响应"""
    from rest_framework.views import exception_handler
    from rest_framework.response import Response

    if isinstance(exc, BusinessError):
        return Response(
            {"error": {"code": exc.code, "message": exc.message}},
            status=exc.status,
        )

    response = exception_handler(exc, context)
    if response is not None:
        response.data = {
            "error": {
                "code": "VALIDATION_ERROR",
                "message": str(exc),
                "details": response.data,
            }
        }
    return response
```

### 步骤 2：使用方式

```python
# ✅ 推荐：使用 ErrorCode 枚举
raise ErrorCode.ROLE_NOT_FOUND.as_exc()

# ✅ 带自定义消息
raise ErrorCode.ROLE_HAS_USERS.as_exc(detail="角色「管理员」下有 5 个用户，不可删除")

# ✅ 校验失败
if User.objects.by_email(email).exists():
    raise ErrorCode.EMAIL_EXISTS.as_exc()

# ❌ 禁止：裸字符串 code
raise BusinessError(message="角色不存在", code="ROLE_NOT_FOUND")
```

### 步骤 3：前端错误处理

```typescript
// front-end/src/utils/error-handler.ts
import { Message } from '@arco-design/web-vue'

const ERROR_MESSAGES: Record<string, string> = {
    USER_NOT_FOUND: '用户不存在',
    EMAIL_EXISTS: '该邮箱已被注册',
    PERMISSION_DENIED: '您没有执行此操作的权限',
    // ... 与后端 ErrorCode.to_frontend_map() 同步
}

export function handleApiError(error: { code: string; message: string; status: number }) {
    switch (error.code) {
        case 'PERMISSION_DENIED':
            router.push('/403')
            break
        case 'INVALID_CREDENTIALS':
            // 登录页特殊处理
            break
        default:
            Message.error(ERROR_MESSAGES[error.code] || error.message || '操作失败')
    }
}
```
